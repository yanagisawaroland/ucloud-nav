GitLab + Gitee 仓库初始化与推送

1. 创建本地目录和.gitkeep

mkdir -p /www/master81-v1/other81-v1/{k3s,cilium,istio}

cd /www/master81-v1/other81-v1

git config user.email "root@devops81.lan"
git config user.name "root"
git init

touch k3s/.gitkeep cilium/.gitkeep istio/.gitkeep

2. 首次提交

git add .
git commit -m "初始化目录结构"

3. 查 GitLab 已有组列表

curl -s "http://gitlab.devops.local:7591/api/v4/groups" \
  -H "PRIVATE-TOKEN: 你的GitLabToken" \
  | python3 -c "import sys,json; print('\n'.join([f'{g[\"id\"]}: {g[\"name\"]}' for g in json.load(sys.stdin)]))"

4. 在 GitLab 创建组 master81-v1

curl -s -X POST "http://gitlab.devops.local:7591/api/v4/groups" \
  -H "PRIVATE-TOKEN: 你的GitLabToken" \
  -H "Content-Type: application/json" \
  -d '{"name":"master81-v1","path":"master81-v1","visibility":"private"}' \
  | python3 -c "import sys,json; d=json.load(sys.stdin); print('组ID:', d.get('id', d.get('message','')))"

5. 在 GitLab 创建私有仓库 other81-v1

curl -s -X POST "http://gitlab.devops.local:7591/api/v4/projects" \
  -H "PRIVATE-TOKEN: 你的GitLabToken" \
  -H "Content-Type: application/json" \
  -d '{"name":"other81-v1","namespace_id":135,"visibility":"private"}' \
  | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('http_url_to_repo', d.get('message','')))"

6. 绑定 GitLab 并推送

cd /www/master81-v1/other81-v1

git remote add gitlab http://root:huzhaofeng1234@gitlab.devops.local:7591/master81-v1/other81-v1.git

git push gitlab master

7. 在 Gitee 创建私有仓库 other81-v1

curl -s -X POST "https://gitee.com/api/v5/orgs/master81-v1/repos" \
  -H "Content-Type: application/json" \
  -d '{"access_token":"cbd8acffa5f4e038b9327d85313dd587","name":"other81-v1","private":true,"auto_init":false}' \
  | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('full_name', d.get('message','')))"

8. 绑定 Gitee 并推送

cd /www/master81-v1/other81-v1

git remote add gitee https://yanagisawa:cbd8acffa5f4e038b9327d85313dd587@gitee.com/master81-v1/other81-v1.git

git push gitee master
