K3s v1.31.5+k3s1 安装配置

主机: master81 (10.20.50.81) / node82 (10.20.50.82)

master81 安装

curl -sfL https://rancher-mirror.rancher.cn/k3s/k3s-install.sh | INSTALL_K3S_MIRROR=cn INSTALL_K3S_VERSION="v1.31.5+k3s1" \
  sh -s - server \
  --disable traefik \
  --disable servicelb \
  --disable-kube-proxy \
  --flannel-backend=none \
  --cluster-cidr=10.42.0.0/16 \
  --service-cidr=10.43.0.0/16 \
  --node-ip=10.20.50.81 \
  --advertise-address=10.20.50.81 \
  --write-kubeconfig-mode=644

node82 加入集群

curl -sfL https://rancher-mirror.rancher.cn/k3s/k3s-install.sh | INSTALL_K3S_MIRROR=cn INSTALL_K3S_VERSION="v1.31.5+k3s1" \
  sh -s - agent \
  --server https://10.20.50.81:6443 \
  --token <TOKEN> \
  --node-ip=10.20.50.82

registries.yaml（两个节点都要配）

路径: /etc/rancher/k3s/registries.yaml

mirrors:
  docker.io:
    endpoint:
      - "http://harbor.devops.local:7581"
  "harbor.devops.local:7581":
    endpoint:
      - "http://harbor.devops.local:7581"
configs:
  "harbor.devops.local:7581":
    auth:
      username: admin
      password: 你的Harbor密码
    tls:
      insecure_skip_verify: true

环境变量

echo 'export KUBECONFIG=/etc/rancher/k3s/k3s.yaml' >> /etc/profile
source /etc/profile
