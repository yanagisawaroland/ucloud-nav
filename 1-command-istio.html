# Istio 核心常用命令
# 1. 查看 Istio 版本
istioctl version

# 2. 查看代理状态（Sidecar 同步情况）
istioctl proxy-status

# 3. 分析集群 Istio 配置/状态问题
istioctl analyze

# 补充常用命令
istioctl install --set profile=default  # 安装 Istio（默认配置）
istioctl uninstall --purge -y           # 彻底卸载 Istio（清理所有资源）
istioctl proxy-config routes  -o yaml  # 查看 Pod 的 Sidecar 路由规则
istioctl describe pod         # 查看 Pod 的 Istio 配置详情
istioctl logs  -c istio-proxy # 查看 Sidecar 日志
